    type OrderListener struct {
    ID          int32
    Detail      string
    OrderNo     string
    OrderAmount float32
    AccountId   int32
    Status      codes.Code
    Receiver    string
    Addr        string
    PostCode    string
    Mobile      string
}

//执行本地事务
func (in *OrderListener) ExecuteLocalTransaction(message *primitive.Message) primitive.LocalTransactionState {
    var orderItem model.OrderItem
    err := json.Unmarshal(message.Body, &orderItem)
    if err != nil {
        zap.S().Error("ExecuteLocalTransaction-Unmarshal-Error" + err.Error())
        in.Detail = "ExecuteLocalTransaction-Unmarshal-Error" + err.Error()
        return primitive.RollbackMessageState
    }
    var productIds []int32
    var carList []model.ShopCart
    productNumMap := make(map[int32]int32)
    r := internal.DB.Where(&model.ShopCart{AccountId: in.AccountId, Checked: true}).Find(&carList)
    if r.RowsAffected == 0 {
        in.Detail = custom_error.ProductNotChecked
        in.OrderAmount = 0
        return primitive.RollbackMessageState
    }
    for _, cart := range carList {
        productIds = append(productIds, cart.ProductId)
        productNumMap[cart.ProductId] = cart.Num
    }
    fmt.Println("productNumMap:", productNumMap)
    fmt.Println("productIds:", productIds)
    productRes, err := internal.ProductClient.BatchGetProduct(context.Background(), &pb.BatchProductIdReq{Ids: productIds})
    if err != nil {
        in.Detail = custom_error.ProductNotFound
        in.OrderAmount = 0
        return primitive.RollbackMessageState
    }
    var amount float32 //总价 = 单价 * 数量
    var orderProductList []model.OrderProduct
    var stockItemList []*pb.ProductStockItem
    for _, p := range productRes.ItemList {
        amount += p.RealPrice * float32(productNumMap[p.Id])
        stockItem := &pb.ProductStockItem{
            ProductId: p.Id,
            Num:       productNumMap[p.Id],
        }
        stockItemList = append(stockItemList, stockItem)
    }
    fmt.Println("stockItemList:", stockItemList)
    //微服务原则：尽量使用List，减少请求次数
    _, err = internal.StockClient.Sell(context.Background(), &pb.SellItem{
        StockItemList: stockItemList,
    })
    if err != nil {
        in.Detail = custom_error.SotckNotEnough
        in.OrderAmount = 0
        return primitive.RollbackMessageState
    }
    tx := internal.DB.Begin()
    orderItem.Status = "unPay"
    orderItem.OrderAmount = amount
    result := tx.Save(&orderItem)
    //事务执行失败，提交回滚
    if result.Error != nil || result.RowsAffected < 1 {
        tx.Rollback()
        in.Detail = custom_error.CreateOrderFail
        in.OrderAmount = 0
        return primitive.CommitMessageState
    }
    for _, p := range productRes.ItemList {
        var orderProduct = model.OrderProduct{
            ProductId:    p.Id,
            ProductName:  p.Name,
            ConvertImage: p.CoverImage,
            RealPrice:    p.RealPrice,
            OrderId:      orderItem.ID,
            Num:          productNumMap[p.Id],
        }
        orderProductList = append(orderProductList, orderProduct)
    }
    fmt.Println("orderProductList:", orderProductList)
    result = tx.CreateInBatches(orderProductList, 50)
    if result.Error != nil || result.RowsAffected < 1 {
        tx.Rollback()
        in.Detail = custom_error.CreateOrderFail
        in.OrderAmount = 0
        return primitive.CommitMessageState
    }
    result = tx.Where(&model.ShopCart{AccountId: in.AccountId, Checked: true}).Delete(&model.ShopCart{})
    if result.Error != nil || result.RowsAffected < 1 {
        tx.Rollback()
        in.Detail = custom_error.CreateOrderFail
        in.OrderAmount = 0
        return primitive.CommitMessageState
    }
    tx.Commit()
    //库存和订单都执行成功
    return primitive.RollbackMessageState
}

    //7.回查本地事务
func (in *OrderListener) CheckLocalTransaction(message *primitive.MessageExt) primitive.LocalTransactionState {
    var orderItem model.OrderItem
    err := json.Unmarshal(message.Body, &orderItem)
    if err != nil {
        zap.S().Error("CheckLocalTransaction-Unmarshal-Error" + err.Error())
        return primitive.UnknowState
    }
    var temp model.OrderItem
    r := internal.DB.Model(&model.OrderItem{}).Where(model.OrderItem{OrderNo: in.OrderNo}).First(temp)
    //订单创建失败，提交回滚(归还库存)
    if r.RowsAffected < 1 {
        return primitive.CommitMessageState
    }
    return primitive.RollbackMessageState
}

    pushConsumer, _ := rocketmq.NewPushConsumer(
    consumer.WithNameServer([]string{"192.168.31.51:9876"}),
    consumer.WithGroupName("StockGroup"),
)
pushConsumer.Subscribe("BackStockTopic",
    consumer.MessageSelector{},
    func(ctx context.Context, me ...*primitive.MessageExt) (consumer.ConsumeResult, error) {
        for i := range me {
            var order model.Order
            err := json.Unmarshal(me[i].Body, &order)
            if err != nil {
                zap.S().Error("Unmarshal Error:" + err.Error())
                return consumer.ConsumeSuccess, nil
            }
            tx := internal.DB.Begin()
            var detail model.StockItemDetail
            r1 := tx.Where(&model.StockItemDetail{
                OrderNo: order.OrderNo,
                Status:  model.HasSell,
            }).First(&detail)
            if r1.RowsAffected < 1 {
                zap.S().Error("StockItemDetail Not Found:" + err.Error())
                return consumer.ConsumeSuccess, nil
            }
            //库存归还
            for _, item := range detail.DetailList {
                r2 := tx.Model(&model.Stock{ProductId: item.ProductId}).Update(
                    "num", gorm.Expr("num+?", item.Num),
                )
                if r2.RowsAffected < 1 {
                    tx.Rollback()
                    return consumer.ConsumeRetryLater, nil
                }
            }
            r3 := tx.Where(&model.StockItemDetail{
                OrderNo: order.OrderNo,
            }).Update("status", model.HasBack)
            if r3.RowsAffected < 1 {
                tx.Rollback()
                return consumer.ConsumeRetryLater, nil
            }
            tx.Commit()
        }
        return consumer.ConsumeSuccess, nil
    })

    micro/cartorder/cartorder_srv/biz/order.go

    //执行本地事务
func (in *OrderListener) ExecuteLocalTransaction(message *primitive.Message) primitive.LocalTransactionState {
    ...
    //发送延迟消息
    p, err := rocketmq.NewProducer(producer.WithNameServer([]string{"192.168.31.51:9876"}))
    if err != nil {
        zap.S().Error("NewProducer Error" + err.Error())
        in.Detail = "NewProducer Error" + err.Error()
        in.Status = codes.Internal
        tx.Rollback()
        return primitive.CommitMessageState
    }
    err = p.Start()
    if err != nil {
        zap.S().Error("Producer Start Error" + err.Error())
        in.Detail = "Producer Start Error" + err.Error()
        in.Status = codes.Internal
        tx.Rollback()
        return primitive.CommitMessageState
    }
    msg := primitive.NewMessage("timeout_order_info", message.Body)
    msg.WithDelayTimeLevel(6)
    _, err = p.SendSync(context.Background(), msg)
    if err != nil {
        zap.S().Error("Producer SendSync Error" + err.Error())
        in.Detail = "Producer SendSync Error" + err.Error()
        in.Status = codes.Internal
        tx.Rollback()
        return primitive.CommitMessageState
    }
    tx.Commit()
    //库存和订单都执行成功
    in.ID = orderItem.ID
    in.OrderAmount = orderItem.OrderAmount
    in.Status = codes.OK
    return primitive.RollbackMessageState
}
