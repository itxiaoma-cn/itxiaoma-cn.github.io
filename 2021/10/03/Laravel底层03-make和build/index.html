    getClosure()ж–№жі•йҖ’еҪ’и°ғз”Ё

    /**
 * з¬¬дёҖж¬Ўи°ғз”Ё
 * $abstract 'Illuminate\Contracts\Http\Kernel'
 * $concrete 'App\Http\Kernel'
 * 
 * з¬¬дәҢж¬Ўи°ғз”Ё
 * иҺ·еҸ–App\Http\Kernelзҡ„е®һдҫӢпјҢеҗ‘дёҠиҝ”еӣһз»ҷbuild()ж–№жі•
 */
protected function getClosure($abstract, $concrete)
{
    return function ($container, $parameters = []) use ($abstract, $concrete) {
        if ($abstract == $concrete) {
            return $container->build($concrete);
        }
        return $container->resolve($concrete, $parameters, $raiseEvents = false);
    };
}

    buildж–№жі•

    vendor/laravel/framework/src/Illuminate/Container/Container.php

    /**
 * з¬¬дёҖж¬Ўи°ғз”Ё
 * $concrete еҢҝеҗҚеҮҪж•°
 * 
 * з¬¬дәҢж¬Ўи°ғз”Ё
 * $concrete 'App\Http\Kernel' 
 * жү§иЎҢеҸҚе°„пјҢиҺ·еҸ–App\Http\Kernelзұ»зҡ„е®һдҫӢеҢ–еҜ№иұЎ
 * иҝ”еӣһз»ҷresolve()ж–№жі•
 */
public function build($concrete)
{
    if ($concrete instanceof Closure) {
        //bind()ж–№жі•дёӯ$concrete = $this->getClosure($abstract, $concrete);
        return $concrete($this, $this->getLastParameterOverride());
    }
    $reflector = new ReflectionClass($concrete);
    if (! $reflector->isInstantiable()) {
        return $this->notInstantiable($concrete);
    }
    $this->buildStack[] = $concrete;
    $constructor = $reflector->getConstructor();
    if (is_null($constructor)) {
        array_pop($this->buildStack);
        return new $concrete;
    }
    $dependencies = $constructor->getParameters();
    $instances = $this->resolveDependencies(
        $dependencies
    );
    array_pop($this->buildStack);
    return $reflector->newInstanceArgs($instances);
}
